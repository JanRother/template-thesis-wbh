% STYLE SETTINGS



% tabularray settings
% custom table environment
\NewTblrEnviron{university-table}
\SetTblrInner[university-table]{
    hlines,
    vlines,
    %row{odd} = {whiteSmoke},				% enable for alternating row colors
	% header settings
    row{1} = {bg=universityBlue, fg=white}
}

\NewTblrTheme{university}{
	% german translations
	\DefTblrTemplate{contfoot-text}{default}{(Fortsetzung auf der n√§chsten Seite.)}
	\DefTblrTemplate{conthead-text}{default}{(Fortsetzung von der vorherigen Seite.)}

	% place caption below longtable
	\DefTblrTemplate{firsthead,middlehead,lasthead}{default}{}
	\SetTblrStyle{contfoot}{halign=c}
	\SetTblrStyle{conthead}{halign=c}
	\DefTblrTemplate{firstfoot}{default}{
		\UseTblrTemplate{caption}{default}
		\UseTblrTemplate{contfoot}{default}
	}
	\DefTblrTemplate{middlefoot}{default}{
		\UseTblrTemplate{capcont}{default}
		\UseTblrTemplate{contfoot}{default}
	}
	\DefTblrTemplate{lastfoot}{default}{
		\UseTblrTemplate{note}{default}
		\UseTblrTemplate{remark}{default}
		\UseTblrTemplate{caption}{default}
		\UseTblrTemplate{conthead}{default}
	}
}

% custom long table environment
\NewTblrEnviron{university-long-table}
\SetTblrOuter[university-long-table]{
	long,
	% theme settings
	theme = university,
}
\SetTblrInner[university-long-table]{
    hlines,
    vlines,
    row{odd} = {whiteSmoke},
	% header settings
    row{1} = {bg=universityPink, fg=white},
    rowhead = 1,
}
